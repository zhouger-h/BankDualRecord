▶ START — 开始录制（file_name 可选）

{ "cmd": "START", "params": { "task_id": "T20260331001", "cust_name": "张三", "file_name": "T20260331001_张三.mp4" } }

// 响应（file_name 原样回显；未传时为 null）
{ "code": 0, "msg": "录制已启动", "task_id": "T20260331001", "file_name": "T20260331001_张三.mp4" }

// 主动事件（摄像头就绪后推送，含最终文件完整路径）
{ "type": "event", "event": "RECORD_STARTED", "data": { "task_id": "T20260331001", "file_path": "D:\\DualRecord\\T20260331001_张三.mp4" } }

{ "cmd": "SHOW_WINDOW", "params": { "file_name": "T20260331001_张三.mp4" } }

// 响应
{ "code": 0, "msg": "主界面已显示", "file_name": "T20260331001_张三.mp4" }

{ "cmd": "HIDE_WINDOW", "params": {} }
// 响应: { "code": 0, "msg": "主界面已隐藏" }
// 副作用: 自动清除 m_pendingFileName，下次吊起需重新传入 file_name
⏸ PAUSE / ▶▶ RESUME — 暂停 / 继续
发送: { "cmd": "PAUSE",  "params": {} }
发送: { "cmd": "RESUME", "params": {} }
返回: { "code": 0, "msg": "录制已暂停" }  /  { "code": 0, "msg": "录制已继续" }
⏹ STOP — 停止录制（自动关闭摄像头 + 入上传队列）
发送: { "cmd": "STOP", "params": {} }
返回: { "code": 0, "msg": "录制已停止", "file_path": "D:\\DualRecord\\REC_T001_xxx.mp4" }
事件推送: { "type": "event", "event": "RECORD_STOPPED", "data": { "task_id":"...", "file_path":"..." } }
▶ PLAY — 播放指定视频（弹出独立播放器窗口）
发送: { "cmd": "PLAY", "params": { "file_path": "D:\\DualRecord\\REC_xxx.mp4" } }
返回: { "code": 0, "msg": "播放器已启动", "file_path": "..." }

🔧 C++ 集成示例（柜面端）
// 1. 连接控件服务
QTcpSocket *sock = new QTcpSocket;
sock->connectToHost("127.0.0.1", 9527);

// 2. 封帧发送
auto send = [&](const QJsonObject &cmd){
    QByteArray body = QJsonDocument(cmd).toJson(QJsonDocument::Compact);
    quint32 len = body.size();
    QByteArray frame(4, 0);
    frame[0]=(len>>24)&0xFF; frame[1]=(len>>16)&0xFF;
    frame[2]=(len>>8 )&0xFF; frame[3]=len&0xFF;
    sock->write(frame + body);
};

// 3. 吊起界面并预设文件名（用户点「开始录制」将使用该文件名）
send({{"cmd","SHOW_WINDOW"},{"params",QJsonObject{{"file_name","T001_张三.mp4"}}}});

// 4. 或直接通过 START 指定文件名（同时触发录制）
send({{"cmd","START"},{"params",QJsonObject{{"task_id","T001"},{"cust_name","张三"},{"file_name","T001_张三.mp4"}}}});

// 5. 停止录制
send({{"cmd","STOP"},{"params",QJsonObject{}}});

// 6. 播放指定文件
send({{"cmd","PLAY"},{"params",QJsonObject{{"file_path","D:\\DualRecord\\T001_张三.mp4"}}}});
